621 template<typename document_view_iterator_type>
622 MONGOCXX_INLINE bsoncxx::stdx::optional<result::insert_many> collection::insert_many(
623  document_view_iterator_type begin,
624  document_view_iterator_type end,
625  const options::insert& options
626 ) {
627  class bulk_write writes(false);
628 
629  std::map<std::size_t, bsoncxx::document::element> inserted_ids{};
630  std::size_t index = 0;
631  std::for_each(begin, end, [&](const bsoncxx::document::view& current){
632  // TODO: put this somewhere else not in header scope (bsoncxx::builder)
633  if ( !current["_id"]) {
634  bsoncxx::builder::stream::document new_document;
635  new_document << "_id" << bsoncxx::oid(bsoncxx::oid::init_tag)
636  << bsoncxx::builder::stream::concatenate{current};
637 
638  writes.append(model::insert_one(new_document.view()));
639 
640  inserted_ids.emplace(index++, new_document.view()["_id"]);
641  } else {
642  writes.append(model::insert_one(current));
643 
644  inserted_ids.emplace(index++, current["_id"]);
645  }
646 
647  });
648 
649  if (options.write_concern())
650  writes.write_concern(*options.write_concern());
651  result::bulk_write res(std::move(bulk_write(writes).value()));
652  bsoncxx::stdx::optional<result::insert_many> result(result::insert_many(std::move(res), std::move(inserted_ids)));
653  return result;
654 }
